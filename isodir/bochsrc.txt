# A super simple BOCHSRC.TXT file which boots TTYLINUX .ISO in Bochs 2.4.5+

log: bochsout.log
megs: 1024

# Guest clock speed in terms of emulated instructions per virtual second
cpu: count=1, ips=50000000, reset_on_triple_fault=0

mouse: enabled=0, toggle=f12
speaker: enabled=1, mode=system
display_library: x

magic_break: enabled=1


keyboard: keymap=x11-pc-us.map

ata1: enabled=1, ioaddr1=0x170, ioaddr2=0x370, irq=15

# CD-ROM image
ata1-master:  type=cdrom, path="LudOS.iso", status=inserted
boot: cdrom
